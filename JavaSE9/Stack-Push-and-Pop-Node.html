2.6.3.1 Stack Push and Pop Subsubsection

<Stack Push> ≡ 

// Push an item onto the stack
void push(int item) {
  if (tos == 9)
    System.out.println("Stack is full.");
  else
    stck[++tos] = item;

<Stack Pop> ≡ 

// Pop an item from the stack
int pop() {
  if (tos < 0) {
    System.out.println("Stack underflow.");
    return 0;
  } else
    return stck[tos--];
  }
}
