// At this point, the file is open and can be read.
// The following reads characters until EOF is encountered.
try {
  do {
    i = fin.read();
    if (i != -1) System.out.print ((char) i);
  } while (i != -1);
} catch (IOException e) {
  System.out.println("Error Reading File");
